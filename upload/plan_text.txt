Indicore Originals: Implementation Plan
1.1 Project Summary
Indicore Originals is a modern, mobile-first e-commerce platform selling curated products via Instagram-driven traffic. The MVP supports:
 Cash on Delivery (COD) as the sole payment method.
 Single-step checkout with integrated lead collection.
 Product variants, image management, moderated reviews (registered customers only).
 Guest order tracking (Order ID + Email).
 Manual COD verification.
 Homepage content management, FAQs, and analytics.
Key Differentiators:
 Single login page with role-based redirect (admin/customer).
 Guest checkout with automatic order linking upon registration.
 Purchase-validated reviews (only delivered orders can review).
 Manual COD verification workflow (no OTP/SMS/email dependency).
 Fixed 7-section homepage structure (content-managed, not structural).
 All customer communication handled manually by admin via phone or WhatsApp.

// Database-based login rate limiting - no external dependencies
async function checkLoginRateLimit(email: string): Promise<{ allowed: boolean; remainingMinutes: number }> {
const customer = await db.customers.findUnique({ where: { email } });
if (customer?.locked_until && customer.locked_until > new Date()) {
const remainingMinutes = Math.ceil((customer.locked_until.getTime() - Date.now()) / 60000);
return { allowed: false, remainingMinutes };
}
if (customer && customer.login_attempts >= 5) {
await db.customers.update({
where: { email },
data: { locked_until: new Date(Date.now() + 15 * 60 * 1000) }
});
return { allowed: false, remainingMinutes: 15 };
}
return { allowed: true, remainingMinutes: 0 };
}

ALTER TABLE products ADD CONSTRAINT products_slug_unique UNIQUE (slug);
CREATE OR REPLACE FUNCTION generate_unique_slug(base_slug TEXT)
RETURNS TEXT AS $$
DECLARE
new_slug TEXT;
counter INT := 0;
BEGIN
new_slug := base_slug;
WHILE EXISTS (SELECT 1 FROM products WHERE slug = new_slug) LOOP
counter := counter + 1;
new_slug := base_slug || '-' || counter;
END LOOP;
RETURN new_slug;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION generate_order_number()
RETURNS TEXT AS $$
DECLARE
timestamp_part TEXT;
random_part TEXT;
BEGIN
timestamp_part := EXTRACT(EPOCH FROM NOW())::TEXT;
random_part := UPPER(SUBSTRING(MD5(RANDOM()::TEXT) FROM 1 FOR 6));
RETURN 'ORD-' || timestamp_part || '-' || random_part;
END;
$$ LANGUAGE plpgsql;

// POST /api/orders
async function createOrder(req, res) {
// 1. Validate all fields (Zod schema - consent is optional)
const { email, name, phone, address, cart, consent } = req.body;
// 2. Get COD settings
const { cod_min, cod_max } = await getSettings();
const cartTotal = calculateCartTotal(cart);
// 3. Validate COD range
if (cartTotal < cod_min || cartTotal > cod_max) {
return res.status(400).json({ error: `COD available for orders between ${cod_min} and ${cod_max}` });
}
// 4. Find or create customer (email unique, phone non-unique)
const customer = await db.customers.upsert({
where: { email },
update: { phone, name, address: JSON.stringify(address) },
create: {
email, phone, name, address: JSON.stringify(address),
is_registered: false, role: 'customer', email_consent_given: consent ?? false
}
});
// 5. Create order with transaction
const order = await createOrderTransaction(cart, customer, cartTotal, address);
// 6. Return order number for confirmation page
// NOTE: No email notification is sent. Admin monitors dashboard manually.
return res.json({ success: true, orderNumber: order.order_number });
}

async function checkTrackingRateLimit(ip: string): Promise<{ allowed: boolean }> {
const windowStart = new Date(Date.now() - TRACKING_WINDOW_MS);
const attempts = await db.order_track_attempts.count({
where: {
ip_address: ip,
attempt_time: { gte: windowStart }
}
});
if (attempts >= TRACKING_LIMIT) {
return { allowed: false };
}
await db.order_track_attempts.create({
data: {
ip_address: ip,
attempt_time: new Date(),
expires_at: new Date(Date.now() + 24 * 60 * 60 * 1000)
}
});
return { allowed: true };
}

async function submitReview(req, res) {
const { productId, orderId, rating, title, comment, photo } = req.body;
const customerId = req.user.id;
// 1. Verify delivered order containing this product
const hasDeliveredOrder = await db.orders.findFirst({
where: {

--- PAGE 24 ---
customer_id: customerId,
status: 'delivered',
order_items: { some: { variant: { product_id: productId } } }
}
});
if (!hasDeliveredOrder) {
return res.status(403).json({ error: 'You can only review products you have purchased and received' });
}
// 2. Check if already reviewed
const existingReview = await db.reviews.findFirst({
where: { product_id: productId, customer_id: customerId }
});
if (existingReview) {
return res.status(409).json({ error: 'You have already reviewed this product' });
}
// 3. Create review with status='pending'
const review = await db.reviews.create({
data: {
product_id: productId, customer_id: customerId,
order_id: hasDeliveredOrder.id,
rating, title, comment, photo_url: photo,
status: 'pending'
}
});
// No email notification is sent to the customer or admin.
return res.json({ success: true, review });
}

async function getAnalytics(req, res) {
const { startDate, endDate } = parseDateRange(req.query);

const aov = totalOrders > 0 ? (revenue._sum.cart_total ?? 0) / totalOrders : 0;
return res.json({
business: { totalOrders, revenue: revenue._sum.cart_total, customerCount, aov, subscriberCount },
cod: await getCodMetrics(startDate, endDate),
reviews: await getReviewMetrics(startDate, endDate),
products: await getTopProducts(startDate, endDate)
});
}

Row Level Security (RLS)
 Principle: Least privilege - every table has RLS enabled.
 Service role bypasses RLS for admin server-side operations only.
Critical: Guest INSERT policies - The following tables must have INSERT policies that allow unauthenticated users for guest checkout to
work:
 customers - allow INSERT for guest creation
 orders - allow INSERT for guest order creation
 order_items - allow INSERT for order item creation
 order_track_attempts - allow INSERT for rate limiting
Without these INSERT policies, guest checkout will fail with RLS violations.
Admin Route Protection (Middleware)
 Checks role in customers table on every /admin/* request.
 Redirects to /login if not authenticated or not admin.
